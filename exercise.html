AES Cryptography

⇒ AES Cryptography

from cryptography.fernet import Fernet

# Generate a random encryption key
key = Fernet.generate_key()
            
# Create a Fernet cipher object with the key
cipher_suite = Fernet(key)
            
# Get user input for the text to be encrypted
plaintext = input("Enter the text to be encrypted: ").encode('utf-8')
            
# Encrypt the data
cipher_text = cipher_suite.encrypt(plaintext)
            
print(f"Encryption Key: {key.decode('utf-8')}")
print(f"Encrypted Text: {cipher_text.decode('utf-8')}")
            
# Decrypt the data
decrypted_text = cipher_suite.decrypt(cipher_text)
            
print(f"Decrypted Text: {decrypted_text.decode('utf-8')}")

DES Cryptography

⇒ DES Cryptography

from Crypto.Cipher import DES
from Crypto.Random import get_random_bytes
                                
# Generate a random 8-byte key for DES
key = get_random_bytes(8)
                                
# Create a DES cipher object with the key and using ECB mode
cipher = DES.new(key, DES.MODE_ECB)
                                
# Get user input for the text to be encrypted
plaintext = input("Enter the text to be encrypted: ")
                                
# Ensure the plaintext is a multiple of 8 bytes (DES block size)
if len(plaintext) % 8 != 0:
     padding_length = 8 - len(plaintext) % 8
     plaintext += ' ' * padding_length
                            
plaintext_bytes = plaintext.encode('utf-8')
                            
# Encrypt the data
cipher_text = cipher.encrypt(plaintext_bytes)
                            
print(f"Encryption Key: {key.hex()}")
print(f"Encrypted Text (in hex): {cipher_text.hex()}")
                            
# Decrypt the data
decrypted_bytes = cipher.decrypt(cipher_text)
                            
# Remove any padding added during encryption
decrypted_text = decrypted_bytes.rstrip(b' ').decode('utf-8')
                            
print(f"Decrypted Text: {decrypted_text}")

Substitution Cipher

⇒ Substitution Cipher

import string

# Define the substitution alphabet (key)
def create_substitution_key(shift):
    alphabet = string.ascii_lowercase
    shifted_alphabet = alphabet[shift:] + alphabet[:shift]
    return str.maketrans(alphabet, shifted_alphabet)
                            
# Function to encrypt plaintext
def encrypt(plaintext, shift):
    substitution_key = create_substitution_key(shift)
    return plaintext.translate(substitution_key)
                            
# Function to decrypt ciphertext
def decrypt(ciphertext, shift):
    substitution_key = create_substitution_key(-shift)
    return ciphertext.translate(substitution_key)
                            
# Get user input for the text and shift value
plaintext = input("Enter the text to be encrypted: ").lower()
hift = int(input("Enter the shift value (integer): "))
                            
# Encrypt the data
cipher_text = encrypt(plaintext, shift)
                            
print(f"Encrypted Text: {cipher_text}")
                            
# Decrypt the data
decrypted_text = decrypt(cipher_text, shift)
                            
print(f"Decrypted Text: {decrypted_text}")
